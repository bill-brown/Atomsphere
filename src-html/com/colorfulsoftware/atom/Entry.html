026    /**
027     * This class represents an Atom 1.0 entry element.
028     * 
029     * @see <a
030     *      href="http://www.atomenabled.org/developers/syndication/atom-format-spec.php">Atom
031     *      Syndication Format</a>
032     * @author Bill Brown
033     * 
034     *         <pre>
035     *      atomEntry =
036     *          element atom:entry {
037     *          atomCommonAttributes,
038     *          (atomAuthor*
039     *          &amp; atomCategory*
040     *          &amp; atomContent?
041     *          &amp; atomContributor*
042     *          &amp; atomId
043     *          &amp; atomLink*
044     *          &amp; atomPublished?
045     *          &amp; atomRights?
046     *          &amp; atomSource?
047     *          &amp; atomSummary?
048     *          &amp; atomTitle
049     *          &amp; atomUpdated
050     *          &amp; extensionElement*)
051     *          }
052     * </pre>
053     */

057             * 
058             */
059            private static final long serialVersionUID = 5291388675692200218L;
060            private final AtomEntrySourceAdaptor entryAdaptor;
061            private final Content content;
062            private final Published published;
063            private final Source source;
064            private final Summary summary;
065    
066            private List<String> unboundPrefixes;
067    
068            // use the factory method in the FeedDoc.
069            Entry(Id id, Title title, Updated updated, Rights rights, Content content,
070                            List<Author> authors, List<Category> categories,
071                            List<Contributor> contributors, List<Link> links,
072                            List<Attribute> attributes, List<Extension> extensions,
073                            Published published, Summary summary, Source source)
074                            throws AtomSpecException {
075    
076                    // check for functional requirements here because
077                    // they are all optional for a Source element.
078    
079                    // make sure id is present
080                    if (id == null) {
081                            throw new AtomSpecException(
082                                            "atom:entry elements MUST contain exactly one atom:id element.");
083                    }
084                    // make sure title is present
085                    if (title == null) {
086                            throw new AtomSpecException(
087                                            "atom:entry elements MUST contain exactly one atom:title element.");
088                    }
089                    // make sure updated is present
090                    // it is actually checked further up in the reader because of how
091                    // we store entries.
092    
093                    this.entryAdaptor = new AtomEntrySourceAdaptor(id, title, updated,
094                                    rights, authors, categories, contributors, links, attributes,
095                                    extensions);
096    
097                    if (content == null) {
098                            this.content = null;
099                    } else {
100    
101                            if (content.getAttributes() != null) {
102                                    for (Attribute attr : content.getAttributes()) {
103                                            // check for src attribute
104                                            if (attr.getName().equals("src") && summary == null) {
105                                                    throw new AtomSpecException(
106                                                                    "atom:entry elements MUST contain an atom:summary element if the atom:entry contains an atom:content that has a \"src\" attribute (and is thus empty).");
107                                            }
108                                    }
109                            }
110    
111                            this.content = new Content(content);
112                    }
113    
114                    this.published = (published == null) ? null : new Published(published);
115                    this.source = (source == null) ? null : new Source(source);
116                    this.summary = (summary == null) ? null : new Summary(summary);
117    
118                    // check that the extension prefixes are bound to a namespace
119                    this.unboundPrefixes = new LinkedList<String>();
120    
121                    if (entryAdaptor.getUnboundPrefixes() != null) {
122                            this.unboundPrefixes.addAll(entryAdaptor.getUnboundPrefixes());
123                    }
124    
125                    if (source != null && source.getUnboundPrefixes() != null) {
126                            for (String unboundPrefix : source.getUnboundPrefixes()) {
127                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
128                                            this.unboundPrefixes.add(unboundPrefix);
129                                    }
130                            }
131                    }
132    
133                    if (published != null && updated.getUnboundPrefixes() != null) {
134                            for (String unboundPrefix : updated.getUnboundPrefixes()) {
135                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
136                                            this.unboundPrefixes.add(unboundPrefix);
137                                    }
138                            }
139                    }
140    
141                    if (summary != null && summary.getUnboundPrefixes() != null) {
142                            for (String unboundPrefix : summary.getUnboundPrefixes()) {
143                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
144                                            this.unboundPrefixes.add(unboundPrefix);
145                                    }
146                            }
147                    }
148    
149                    this.unboundPrefixes = (this.unboundPrefixes.size() == 0) ? null
150                                    : this.unboundPrefixes;
151            }
152    
153            Entry(Entry entry) {
154                    this.content = entry.getContent();
155                    this.entryAdaptor = entry.getEntryAdaptor();
156                    this.published = entry.getPublished();
157                    this.source = entry.getSource();
158                    this.summary = entry.getSummary();
159                    this.unboundPrefixes = entry.getUnboundPrefixes();
160            }

163                    return new AtomEntrySourceAdaptor(this.entryAdaptor);
164            }

339            @Override
340            public String toString() {
341                    StringBuilder sb = new StringBuilder("<entry");
342                    sb.append(entryAdaptor);
343    
344                    if (published != null) {
345                            sb.append(published);
346                    }
347                    if (summary != null) {
348                            sb.append(summary);
349                    }
350                    if (source != null) {
351                            sb.append(source);
352                    }
353                    if (content != null) {
354                            sb.append(content);
355                    }
356    
357                    sb.append("</entry>");
358                    return sb.toString();
359            }
