031    /**
032     * This class represents an Atom 1.0 feed element.
033     * 
034     * @see <a
035     *      href="http://www.atomenabled.org/developers/syndication/atom-format-spec.php">Atom
036     *      Syndication Format</a>
037     * @author Bill Brown
038     * 
039     *         <pre>
040     *      atomFeed =
041     *          element atom:feed {
042     *          atomCommonAttributes,
043     *          (atomAuthor*
044     *          &amp; atomCategory*
045     *          &amp; atomContributor*
046     *          &amp; atomGenerator?
047     *          &amp; atomIcon?
048     *          &amp; atomId
049     *          &amp; atomLink*
050     *          &amp; atomLogo?
051     *          &amp; atomRights?
052     *          &amp; atomSubtitle?
053     *          &amp; atomTitle
054     *          &amp; atomUpdated
055     *          &amp; extensionElement*),
056     *          atomEntry*
057     *          }
058     * </pre>
059     * 
060     */
061    public class Feed implements Serializable {
062    
063            /**
064             * 
065             */
066            private static final long serialVersionUID = 8576480736811385690L;
067    
068            private final AtomEntrySourceAdaptor sourceAdaptor;
069            private final Generator generator;
070            private final Icon icon;
071            private final Logo logo;
072            private final Subtitle subtitle;
073            private final List<Entry> entries;
074    
075            // extra processing fields
076            private List<String> unboundPrefixes;
077    
078            // use the factory method in the FeedDoc.
079            Feed(Id id, Title title, Updated updated, Rights rights,
080                            List<Author> authors, List<Category> categories,
081                            List<Contributor> contributors, List<Link> links,
082                            List<Attribute> attributes, List<Extension> extensions,
083                            Generator generator, Subtitle subtitle, Icon icon, Logo logo,
084                            List<Entry> entries) throws AtomSpecException {
085    
086                    // make sure id is present
087                    if (id == null) {
088                            throw new AtomSpecException(
089                                            "atom:feed elements MUST contain exactly one atom:id element.");
090                    }
091                    // make sure title is present
092                    if (title == null) {
093                            throw new AtomSpecException(
094                                            "atom:feed elements MUST contain exactly one atom:title element.");
095                    }
096                    // make sure updated is present
097                    if (updated == null) {
098                            throw new AtomSpecException(
099                                            "atom:feed elements MUST contain exactly one atom:updated element.");
100                    }
101    
102                    this.sourceAdaptor = new AtomEntrySourceAdaptor(id, title, updated,
103                                    rights, authors, categories, contributors, links, attributes,
104                                    extensions);
105    
106                    this.generator = (generator == null) ? null : new Generator(generator);
107                    this.subtitle = (subtitle == null) ? null : new Subtitle(subtitle);
108                    this.icon = (icon == null) ? null : new Icon(icon);
109                    this.logo = (logo == null) ? null : new Logo(logo);
110    
111                    // check that the extension prefixes are bound to a namespace
112                    this.unboundPrefixes = new LinkedList<String>();
113    
114                    if (generator != null && generator.getUnboundPrefixes() != null) {
115                            for (String unboundPrefix : generator.getUnboundPrefixes()) {
116                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
117                                            this.unboundPrefixes.add(unboundPrefix);
118                                    }
119                            }
120                    }
121    
122                    if (subtitle != null && subtitle.getUnboundPrefixes() != null) {
123                            for (String unboundPrefix : subtitle.getUnboundPrefixes()) {
124                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
125                                            this.unboundPrefixes.add(unboundPrefix);
126                                    }
127                            }
128                    }
129    
130                    if (icon != null && icon.getUnboundPrefixes() != null) {
131                            for (String unboundPrefix : icon.getUnboundPrefixes()) {
132                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
133                                            this.unboundPrefixes.add(unboundPrefix);
134                                    }
135                            }
136                    }
137    
138                    if (logo != null && logo.getUnboundPrefixes() != null) {
139                            for (String unboundPrefix : logo.getUnboundPrefixes()) {
140                                    if (getAttribute("xmlns:" + unboundPrefix) == null) {
141                                            this.unboundPrefixes.add(unboundPrefix);
142                                    }
143                            }
144                    }
145    
146                    if (sourceAdaptor.getUnboundPrefixes() != null) {
147                            this.unboundPrefixes.addAll(sourceAdaptor.getUnboundPrefixes());
148                    }
149    
150                    if (entries == null) {
151                            this.entries = null;
152                            if (sourceAdaptor.getAuthors() == null) {
153                                    throw new AtomSpecException(
154                                                    "atom:feed elements MUST contain one or more atom:author elements, unless the atom:entry contains an atom:source element that contains an atom:author element or, in an Atom Feed Document, the atom:feed element contains an atom:author element itself.");
155                            }
156                    } else {
157    
158                            this.entries = new LinkedList<Entry>();
159    
160                            for (Entry entry : entries) {
161                                    // check that the entry unbound prefixes are ok.
162                                    if (entry.getUnboundPrefixes() != null) {
163                                            for (String unboundPrefix : entry.getUnboundPrefixes()) {
164                                                    if (sourceAdaptor
165                                                                    .getAttribute("xmlns:" + unboundPrefix) == null) {
166                                                            this.unboundPrefixes.addAll(entry
167                                                                            .getUnboundPrefixes());
168                                                    }
169                                            }
170                                    }
171                                    // if there is no author element at the feed level
172                                    // check to make sure the entry has an author element
173                                    if (sourceAdaptor.getAuthors() == null
174                                                    && entry.getAuthors() == null) {
175                                            throw new AtomSpecException(
176                                                            "atom:feed elements MUST contain one or more atom:author elements, unless all of the atom:feed element's child atom:entry elements contain at least one atom:author element.");
177    
178                                    }
179    
180                                    this.entries.add(new Entry(entry));
181                            }
182                    }
183    
184                    // if there are any unbound prefixes, throw an exception
185                    if (this.unboundPrefixes.size() > 0) {
186                            StringBuilder sb = new StringBuilder();
187                            for (String namePrefix : this.unboundPrefixes) {
188                                    sb.append(namePrefix + " ");
189                            }
190                            throw new AtomSpecException(
191                                            "the following extension prefix(es) ( "
192                                                            + sb
193                                                            + ") are not bound to a namespace declaration. See http://www.w3.org/TR/1999/REC-xml-names-19990114/#ns-decl.");
194                    }
195    
196            }

198            /**
199             * @return the list of entry elements. Sorted by updated descending by
200             *         default.
201             */
202            public List<Entry> getEntries() {
203                    if (entries == null) {
204                            return null;
205                    }
206                    List<Entry> entriesCopy = new LinkedList<Entry>();
207                    for (Entry entry : this.entries) {
208                            entriesCopy.add(new Entry(entry));
209                    }
210                    return entriesCopy;
211            }

362             * @param entryTitle
363             *            the title of this entry
364             * @return the entry with the given title or null if not found.
365             */
366            public Entry getEntry(String entryTitle) {
367                    if (this.entries != null) {
368                            for (Entry entry : this.entries) {
369                                    if (entry.getTitle().getText().equals(entryTitle)) {
370                                            return new Entry(entry);
371                                    }
372                            }
373                    }
374                    return null;
375            }

399            /**
400             * Shows the contents of the &lt;feed> element.
401             */
402            @Override
403            public String toString() {
404                    StringBuilder sb = new StringBuilder("<feed");
405                    sb.append(sourceAdaptor);
406    
407                    if (generator != null) {
408                            sb.append(generator);
409                    }
410    
411                    if (subtitle != null) {
412                            sb.append(subtitle);
413                    }
414    
415                    if (icon != null) {
416                            sb.append(icon);
417                    }
418    
419                    if (logo != null) {
420                            sb.append(logo);
421                    }
422    
423                    if (entries != null) {
424                            for (Entry entry : entries) {
425                                    sb.append(entry);
426                            }
427                    }
428    
429                    sb.append("</feed>");
430                    return sb.toString();
431            }
