019    package com.colorfulsoftware.atom;
020    
021    import java.io.Serializable;
022    import java.util.LinkedList;
023    import java.util.List;
024    
025    /**
026     * This class represents an Atom 1.0 category.
027     * 
028     * @see <a
029     *      href="http://www.atomenabled.org/developers/syndication/atom-format-spec.php">Atom
030     *      Syndication Format</a>
031     * @author Bill Brown
032     * 
033     *         <pre>
034     *      atomCategory =
035     *          element atom:category {
036     *          atomCommonAttributes,
037     *          attribute term { text },
038     *          attribute scheme { atomUri }?,
039     *          attribute label { text }?,
040     *          undefinedContent
041     * </pre>
042     */
043    public class Category implements Serializable {
044    
045            /**
046             * 
047             */
048            private static final long serialVersionUID = -7475777621616906621L;
049            private final List<Attribute> attributes;
050            private final Attribute term; // required
051            private final Attribute scheme;
052            private final Attribute label;
053            private final String content;
054            private List<String> unboundPrefixes = null;
055    
056            // use the factory method in the FeedDoc.
057            Category(List<Attribute> attributes, String content)
058                            throws AtomSpecException {
059    
060                    this.unboundPrefixes = new LinkedList<String>();
061    
062                    if (attributes == null) {
063                            throw new AtomSpecException(
064                                            "Category elements MUST have a \"term\" attribute.");
065                    } else {
066                            this.attributes = new LinkedList<Attribute>();
067                            for (Attribute attr : attributes) {
068                                    // check for unsupported attribute.
069                                    if (!new AttributeSupport(attr).verify(this)) {
070                                            throw new AtomSpecException("Unsupported attribute "
071                                                            + attr.getName() + " in the atom:category element.");
072                                    }
073                                    this.attributes.add(new Attribute(attr));
074                                    // check for unbound attribute prefixes
075                                    if (attr.getName().indexOf(":") != -1
076                                                    && !attr.getName().equals("xml:lang")
077                                                    && !attr.getName().equals("xml:base")
078                                                    && !attr.getName().startsWith("xmlns:")
079                                                    && getAttribute("xmlns:"
080                                                                    + attr.getName().substring(0,
081                                                                                    attr.getName().indexOf(":"))) == null) {
082                                            this.unboundPrefixes.add(attr.getName().substring(0,
083                                                            attr.getName().indexOf(":")));
084                                    }
085                            }
086                    }
087    
088                    this.unboundPrefixes = (this.unboundPrefixes.size() == 0) ? null
089                                    : this.unboundPrefixes;
090    
091                    if ((this.term = getAttribute("term")) == null
092                                    || this.term.getValue().equals("")) {
093                            throw new AtomSpecException(
094                                            "Category term attribue SHOULD NOT be blank.");
095                    }
096    
097                    this.scheme = getAttribute("scheme");
098    
099                    this.label = getAttribute("label");
100    
101                    this.content = (content == null || content.equals("")) ? null : content;
102            }
103    
104            Category(Category category) {
105                    this.attributes = category.getAttributes();
106                    this.term = category.getTerm();
107                    this.scheme = category.getScheme();
108                    this.label = category.getLabel();
109                    this.content = category.content;
110            }
111    
112            /**
113             * 
114             * @return the category attribute list.
115             */
116            public List<Attribute> getAttributes() {
117    
118                    List<Attribute> attrsCopy = new LinkedList<Attribute>();
119                    for (Attribute attr : this.attributes) {
120                            attrsCopy.add(new Attribute(attr));
121                    }
122                    return attrsCopy;
123            }
124    
125            /**
126             * 
127             * @return the label attribute
128             */
129            public Attribute getLabel() {
130                    return (label == null) ? null : new Attribute(label);
131            }
132    
133            /**
134             * 
135             * @return the scheme attribute
136             */
137            public Attribute getScheme() {
138                    return (scheme == null) ? null : new Attribute(scheme);
139            }
140    
141            /**
142             * 
143             * @return the term attribute
144             */
145            public Attribute getTerm() {
146                    return new Attribute(term);
147            }
148    
149            /**
150             * 
151             * @return undefined text content or undefined element.
152             */
153            public String getContent() {
154                    return content;
155            }
156    
157            /**
158             * @param attrName
159             *            the name of the attribute to get.
160             * @return the Attribute object if attrName matches or null if not found.
161             */
162            public Attribute getAttribute(String attrName) {
163                    for (Attribute attribute : this.attributes) {
164                            if (attribute.getName().equals(attrName)) {
165                                    return new Attribute(attribute);
166                            }
167                    }
168                    return null;
169            }
170    
171            /**
172             * Shows the contents of the &lt;category> element.
173             */
174            @Override
175            public String toString() {
176                    StringBuilder sb = new StringBuilder("<category");
177    
178                    for (Attribute attribute : attributes) {
179                            sb.append(attribute);
180                    }
181    
182                    if (content == null) {
183                            sb.append(" />");
184                    } else {
185                            sb.append(" >" + content + "</category>");
186                    }
187    
188                    return sb.toString();
189            }
190    
191            List<String> getUnboundPrefixes() {
192                    return unboundPrefixes;
193            }
194            
195            @Override
196            public boolean equals(Object obj) {
197                    if (obj == this) {
198                            return true;
199                    }
200                    if (!(obj instanceof Category)) {
201                            return false;
202                    }
203                    return this.toString().equals(obj.toString());
204            }
205            
206            @Override public int hashCode() {
207                    return toString().hashCode();
208            }
209    }
