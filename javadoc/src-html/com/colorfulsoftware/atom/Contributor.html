020    package com.colorfulsoftware.atom;
021    
022    import java.io.Serializable;
023    import java.util.List;
024    
025    /**
026     * This class represents an Atom 1.0 Contributor element.
027     * 
028     * @see <a
029     *      href="http://www.atomenabled.org/developers/syndication/atom-format-spec.php">Atom
030     *      Syndication Format</a>
031     * @author Bill Brown
032     * 
033     *         <pre>
034     *      atomContributor = element atom:contributor { atomPersonConstruct }
035     * </pre>
036     */
037    public class Contributor implements Serializable {
038    
039            /**
040             * 
041             */
042            private static final long serialVersionUID = -57255552816934722L;
043            private final AtomPersonConstruct person;
044            private List<String> unboundPrefixes = null;
045    
046            // use the factory method in the FeedDoc.
047            Contributor(Name name, URI uri, Email email, List<Attribute> attributes,
048                            List<Extension> extensions) throws AtomSpecException {
049                    this.person = new AtomPersonConstruct(name, uri, email, attributes,
050                                    extensions);
051                    this.unboundPrefixes = person.getUnboundPrefixes();
052            }
053    
054            Contributor(Contributor contributor) {
055                    this.person = new AtomPersonConstruct(contributor.person);
056            }
057    
058            /**
059             * @return the name element.
060             */
061            public Name getName() {
062                    return person.getName();
063            }
064    
065            /**
066             * @return the uri element.
067             */
068            public URI getUri() {
069                    return person.getUri();
070            }
071    
072            /**
073             * @return the email element.
074             */
075            public Email getEmail() {
076                    return person.getEmail();
077            }
078    
079            /**
080             * @return the list of attributes
081             */
082            public List<Attribute> getAttributes() {
083                    return person.getAttributes();
084            }
085    
086            /**
087             * @return the list of extensions
088             */
089            public List<Extension> getExtensions() {
090                    return person.getExtensions();
091            }
092    
093            /**
094             * @param attrName
095             *            the name of the attribute to get.
096             * @return the Attribute object if attrName matches or null if not found.
097             */
098            public Attribute getAttribute(String attrName) {
099                    return person.getAttribute(attrName);
100            }
101    
102            /**
103             * @param extName
104             *            the element name of the extension to get.
105             * @return the Extension object if extName matches or null if not found.
106             */
107            public Extension getExtension(String extName) {
108                    return person.getExtension(extName);
109            }
110    
111            /**
112             * Shows the contents of the &lt;contributor> element.
113             */
114            @Override
115            public String toString() {
116                    return "<contributor" + person + "</contributor>";
117            }
118    
119            List<String> getUnboundPrefixes() {
120                    return unboundPrefixes;
121            }
122            
123            @Override
124            public boolean equals(Object obj) {
125                    if (obj == this) {
126                            return true;
127                    }
128                    if (!(obj instanceof Contributor)) {
129                            return false;
130                    }
131                    return this.toString().equals(obj.toString());
132            }
133            
134            @Override public int hashCode() {
135                    return toString().hashCode();
136            }
137    }
